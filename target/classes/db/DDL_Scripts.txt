CREATE DATABASE `myjpa` /*!40100 DEFAULT CHARACTER SET utf8 */$$

delimiter $$

CREATE TABLE `client` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `name` varchar(45) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=3 DEFAULT CHARSET=utf8$$


delimiter $$

CREATE TABLE `client_answer` (
  `client_id` int(11) NOT NULL,
  `question_id` int(11) NOT NULL,
  `answer_type` varchar(4) NOT NULL,
  PRIMARY KEY (`client_id`,`question_id`,`answer_type`),
  KEY `fk_question` (`question_id`),
  CONSTRAINT `fk_client` FOREIGN KEY (`client_id`) REFERENCES `client` (`id`) ON DELETE NO ACTION ON UPDATE NO ACTION,
  CONSTRAINT `fk_question` FOREIGN KEY (`question_id`) REFERENCES `question_answer` (`question_id`) ON DELETE NO ACTION ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$


delimiter $$

CREATE TABLE `question` (
  `question_id` int(11) NOT NULL AUTO_INCREMENT,
  `question_details` varchar(45) NOT NULL,
  PRIMARY KEY (`question_id`)
) ENGINE=InnoDB AUTO_INCREMENT=4 DEFAULT CHARSET=utf8$$


delimiter $$

CREATE TABLE `question_answer` (
  `question_id` int(11) NOT NULL,
  `answer_type` varchar(4) NOT NULL,
  PRIMARY KEY (`question_id`,`answer_type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8$$
